### 2.2 实现核心算法

**algorithm-service/infer.py**：
```python
import cv2
import numpy as np
from pathlib import Path

class CitrusGrader:
    """传统 CV 柑橘分级器（无深度学习）"""
    
    def __init__(self):
        # 分级标准（砂糖橘）
        self.min_diameter = 45  # mm
        self.color_th1 = 85     # 一级着色率阈值
        self.color_th2 = 70     # 二级着色率阈值
    
    def segment_fruit(self, img):
        """果实分割（简单阈值法）"""
        # 转换到 HSV
        hsv = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)
        
        # 橙色范围（H: 10-25, S: 100-255, V: 100-255）
        lower = np.array([10, 100, 100])
        upper = np.array([25, 255, 255])
        mask = cv2.inRange(hsv, lower, upper)
        
        # 形态学处理
        kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (15, 15))
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        
        return mask
    
    def estimate_diameter(self, mask):
        """估计果径（mm）"""
        # 查找轮廓
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        if not contours:
            return 0
        
        # 最大轮廓的最小外接圆
        largest = max(contours, key=cv2.contourArea)
        (x, y), radius = cv2.minEnclosingCircle(largest)
        
        # 假设图片宽度对应实际 10cm（需标定，这里简化）
        pixel_to_mm = 100 / mask.shape[1]
        diameter_mm = radius * 2 * pixel_to_mm
        
        return round(diameter_mm, 1)
    
    def calculate_color_ratio(self, img, mask):
        """计算着色率（%）"""
        hsv = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)
        
        # 橙色范围（成熟颜色）
        lower_orange = np.array([10, 100, 100])
        upper_orange = np.array([25, 255, 255])
        orange_mask = cv2.inRange(hsv, lower_orange, upper_orange)
        
        # 果实区域内的着色像素
        fruit_area = cv2.countNonZero(mask)
        colored_area = cv2.countNonZero(cv2.bitwise_and(orange_mask, mask))
        
        if fruit_area == 0:
            return 0
        
        ratio = (colored_area / fruit_area) * 100
        return round(ratio, 1)
    
    def grade(self, diameter_mm, color_ratio):
        """规则分级"""
        # 一级：果径达标 且 着色率>=85%
        if diameter_mm >= self.min_diameter and color_ratio >= self.color_th1:
            return "grade1"
        
        # 二级：着色率>=70%
        elif color_ratio >= self.color_th2:
            return "grade2"
        
        # 等外品
        else:
            return "out"
    
    def infer(self, image_path):
        """完整推理流程"""
        # 读取图片
        img = cv2.imread(str(image_path))
        if img is None:
            raise ValueError(f"无法读取图片: {image_path}")
        
        # 1. 果实分割
        mask = self.segment_fruit(img)
        
        # 2. 估计果径
        diameter_mm = self.estimate_diameter(mask)
        
        # 3. 计算着色率
        color_ratio = self.calculate_color_ratio(img, mask)
        
        # 4. 分级
        grade = self.grade(diameter_mm, color_ratio)
        
        # 5. 生成标注图（在果实上画圆）
        annotated = img.copy()
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        if contours:
            largest = max(contours, key=cv2.contourArea)
            (x, y), radius = cv2.minEnclosingCircle(largest)
            cv2.circle(annotated, (int(x), int(y)), int(radius), (0, 255, 0), 3)
        
        annotated_path = Path(image_path).parent / f"{Path(image_path).stem}_annotated.jpg"
        cv2.imwrite(str(annotated_path), annotated)
        
        return {
            "grade": grade,
            "diameter_mm": diameter_mm,
            "color_ratio": color_ratio,
            "defect_ratio": 0,  # 传统CV不检测缺陷，填0
            "defect_count": 0,
            "defects": [],
            "annotated_path": str(annotated_path),
            "confidence": 0.85,  # 固定置信度
            "is_fallback": False
        }

# 测试
if __name__ == "__main__":
    grader = CitrusGrader()
    result = grader.infer("test.jpg")
    print(result)
```

### 2.3 封装为 FastAPI 服务

**algorithm-service/server.py**：
```python
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from infer import CitrusGrader

app = FastAPI()
grader = CitrusGrader()

class InferRequest(BaseModel):
    image_path: str
    standard_id: int = 1

@app.post("/infer")
def infer(req: InferRequest):
    try:
        result = grader.infer(req.image_path)
        return result
    except Exception as e:
        raise HTTPException(500, str(e))

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8001)
```

### 4.2 实现核心接口

**backend/app/main.py**（简化版）：
```python
from fastapi import FastAPI, UploadFile, File
from fastapi.middleware.cors import CORSMiddleware
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker
import httpx
import uuid
from pathlib import Path
from datetime import datetime

app = FastAPI()

# CORS（允许前端跨域）
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"]
)

# 数据库连接
DATABASE_URL = "mysql+pymysql://root:root@localhost:3306/citrus_vision"
engine = create_engine(DATABASE_URL)
Session = sessionmaker(bind=engine)

# 上传目录
UPLOAD_DIR = Path("../uploads")
UPLOAD_DIR.mkdir(exist_ok=True)

# ========== 接口1：上传图片 ==========
@app.post("/api/v1/samples")
async def upload_sample(file: UploadFile = File(...)):
    # 生成唯一文件名
    file_id = uuid.uuid4().hex[:8]
    filename = f"{file_id}_{file.filename}"
    file_path = UPLOAD_DIR / filename
    
    # 保存文件
    with open(file_path, "wb") as f:
        f.write(await file.read())
    
    # 插入数据库
    db = Session()
    db.execute(
        "INSERT INTO product_sample (image_path, created_at) VALUES (%s, %s)",
        (str(file_path), datetime.now())
    )
    db.commit()
    sample_id = db.execute("SELECT LAST_INSERT_ID()").scalar()
    db.close()
    
    return {"sample_id": sample_id, "file_path": str(file_path)}

# ========== 接口2：创建检测任务 ==========
@app.post("/api/v1/tasks")
async def create_task(sample_id: int):
    db = Session()
    
    # 创建任务
    task_no = f"TASK-{uuid.uuid4().hex[:8]}"
    db.execute(
        "INSERT INTO detection_task (task_no, user_id, status, total_count, done_count) VALUES (%s, 1, 'processing', 1, 0)",
        (task_no,)
    )
    db.commit()
    task_id = db.execute("SELECT LAST_INSERT_ID()").scalar()
    
    # 获取图片路径
    sample = db.execute("SELECT image_path FROM product_sample WHERE id=%s", (sample_id,)).fetchone()
    image_path = sample[0]
    
    # 调用算法服务
    async with httpx.AsyncClient() as client:
        resp = await client.post(
            "http://localhost:8001/infer",
            json={"image_path": image_path}
        )
        result = resp.json()
    
    # 保存结果
    db.execute(
        """INSERT INTO detection_result (task_id, sample_id, grade, diameter_mm, color_ratio, defect_ratio, defect_count, annotated_path, confidence)
           VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s)""",
        (task_id, sample_id, result["grade"], result["diameter_mm"], result["color_ratio"], 
         result["defect_ratio"], result["defect_count"], result["annotated_path"], result["confidence"])
    )
    db.execute("UPDATE detection_task SET status='done', done_count=1 WHERE id=%s", (task_id,))
    db.commit()
    
    result_id = db.execute("SELECT LAST_INSERT_ID()").scalar()
    db.close()
    
    return {"task_id": task_id, "result_id": result_id}

# ========== 接口3：获取结果 ==========
@app.get("/api/v1/results/{result_id}")
def get_result(result_id: int):
    db = Session()
    result = db.execute(
        "SELECT grade, diameter_mm, color_ratio, defect_ratio, annotated_path FROM detection_result WHERE id=%s",
        (result_id,)
    ).fetchone()
    db.close()
    
    if not result:
        return {"error": "结果不存在"}
    
    return {
        "grade": result[0],
        "diameter_mm": result[1],
        "color_ratio": result[2],
        "defect_ratio": result[3],
        "annotated_url": f"/uploads/{Path(result[4]).name}"
    }

# 静态文件（标注图）
from fastapi.staticfiles import StaticFiles
app.mount("/uploads", StaticFiles(directory="../uploads"), name="uploads")

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

## 🎨 Day 6：前端页面（黄浩然）

**只做2个页面**。

### 6.1 上传页（Upload.vue）

```vue
<template>
  <div class="upload-page">
    <h1>柑橘品质检测</h1>
    
    <input type="file" @change="onFileSelect" accept="image/*">
    <button @click="startDetection" :disabled="!file || loading">
      {{ loading ? '检测中...' : '开始检测' }}
    </button>
    
    <div v-if="result">
      <h2>检测结果</h2>
      <p>等级：{{ gradeMap[result.grade] }}</p>
      <p>果径：{{ result.diameter_mm }} mm</p>
      <p>着色率：{{ result.color_ratio }}%</p>
      <img :src="result.annotated_url" alt="标注图">
    </div>
  </div>
</template>

<script setup>
import { ref } from 'vue'
import axios from 'axios'

const API_BASE = 'http://localhost:8000/api/v1'

const file = ref(null)
const loading = ref(false)
const result = ref(null)

const gradeMap = {
  'grade1': '一级（精品）',
  'grade2': '二级（合格）',
  'out': '等外品'
}

const onFileSelect = (e) => {
  file.value = e.target.files[0]
}

const startDetection = async () => {
  loading.value = true
  
  try {
    // 1. 上传图片
    const formData = new FormData()
    formData.append('file', file.value)
    const uploadResp = await axios.post(`${API_BASE}/samples`, formData)
    const sampleId = uploadResp.data.sample_id
    
    // 2. 创建检测任务
    const taskResp = await axios.post(`${API_BASE}/tasks?sample_id=${sampleId}`)
    const resultId = taskResp.data.result_id
    
    // 3. 获取结果
    const resultResp = await axios.get(`${API_BASE}/results/${resultId}`)
    result.value = resultResp.data
  } catch (err) {
    alert('检测失败：' + err.message)
  } finally {
    loading.value = false
  }
}
</script>

<style scoped>
.upload-page {
  max-width: 600px;
  margin: 50px auto;
  text-align: center;
}
button {
  margin-top: 20px;
  padding: 10px 30px;
}
img {
  max-width: 100%;
  margin-top: 20px;
}
</style>
```
